
const SPREADSHEET_ID = "여기에_구글시트_ID_입력";
const REQUEST_SHEET = "LeaveRequests";
const EMPLOYEE_SHEET = "Employees";
const ADMIN_PASSWORD = "1234";
const TIMEZONE = "Asia/Seoul";

function doGet(e) {
  const p = e.parameter || {};
  const action = p.action || "";
  const callback = p.callback || "";
  let result = { ok:false, message:"unknown action" };
  try {
    setupSheets_();
    if (action === "list") {
      if (p.password !== ADMIN_PASSWORD) throw new Error("관리자 비밀번호가 다릅니다.");
      result = { ok:true, rows:getRequests_() };
    } else if (action === "my") {
      result = { ok:true, rows:getMyRequests_(p.name || "", p.phone || "") };
    } else if (action === "balance") {
      result = { ok:true, balance:getBalance_(p.name || "", p.phone || "") };
    } else {
      result = { ok:false, message:"알 수 없는 요청입니다." };
    }
  } catch (err) {
    result = { ok:false, message:err.message };
  }
  const out = JSON.stringify(result);
  if (callback) return ContentService.createTextOutput(callback + "(" + out + ")").setMimeType(ContentService.MimeType.JAVASCRIPT);
  return ContentService.createTextOutput(out).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  let result = { ok:false, message:"처리 실패" };
  try {
    setupSheets_();
    const data = JSON.parse(e.postData.contents || "{}");
    if (data.action === "register") result = registerEmployee_(data);
    else if (data.action === "apply") result = applyLeave_(data);
    else if (data.action === "approve") {
      if (data.password !== ADMIN_PASSWORD) throw new Error("관리자 비밀번호가 다릅니다.");
      result = updateStatus_(data.id, "승인", data.adminMemo || "");
    } else if (data.action === "reject") {
      if (data.password !== ADMIN_PASSWORD) throw new Error("관리자 비밀번호가 다릅니다.");
      result = updateStatus_(data.id, "반려", data.adminMemo || "");
    } else throw new Error("알 수 없는 요청입니다.");
  } catch (err) {
    result = { ok:false, message:err.message };
  }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function setupSheets_() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let req = ss.getSheetByName(REQUEST_SHEET);
  if (!req) {
    req = ss.insertSheet(REQUEST_SHEET);
    req.appendRow(["ID","신청일시","매장","이름","연락처","휴가종류","시작일","종료일","사용일수","사유","상태","관리자메모","처리일시"]);
  }
  let emp = ss.getSheetByName(EMPLOYEE_SHEET);
  if (!emp) {
    emp = ss.insertSheet(EMPLOYEE_SHEET);
    emp.appendRow(["등록일시","매장","이름","연락처","입사일","비고"]);
  }
}

function registerEmployee_(data) {
  const store = String(data.store || "한국의집 롯데월드몰").trim();
  const name = String(data.name || "").trim();
  const phone = normalizePhone_(data.phone || "");
  const hireDate = String(data.hireDate || "").trim();
  if (!name) throw new Error("이름을 입력하세요.");
  if (!phone) throw new Error("연락처를 입력하세요.");
  if (!hireDate) throw new Error("입사일을 입력하세요.");
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sh = ss.getSheetByName(EMPLOYEE_SHEET);
  const rows = sh.getDataRange().getValues();
  for (let i=1;i<rows.length;i++) {
    if (String(rows[i][2]).trim() === name && normalizePhone_(rows[i][3]) === phone) {
      sh.getRange(i+1,2).setValue(store);
      sh.getRange(i+1,5).setValue(hireDate);
      return { ok:true, message:"직원 정보가 수정되었습니다." };
    }
  }
  sh.appendRow([new Date(), store, name, phone, hireDate, ""]);
  return { ok:true, message:"직원 등록이 완료되었습니다." };
}

function applyLeave_(data) {
  const name = String(data.name || "").trim();
  const phone = normalizePhone_(data.phone || "");
  const store = String(data.store || "한국의집 롯데월드몰").trim();
  const leaveType = String(data.leaveType || "").trim();
  const startDate = String(data.startDate || "").trim();
  const endDate = String(data.endDate || "").trim();
  const days = Number(data.days || 0);
  const reason = String(data.reason || "").trim();
  if (!name || !phone || !leaveType || !startDate || !endDate || !days) throw new Error("필수 항목을 확인하세요.");
  if (!findEmployee_(name, phone)) throw new Error("직원 등록 정보가 없습니다. 먼저 직원등록을 해주세요.");
  const balance = getBalance_(name, phone);
  if ((leaveType === "연차" || leaveType === "월차" || leaveType.indexOf("반차") >= 0) && days > balance.remain) {
    throw new Error("잔여 연월차가 부족합니다. 현재 잔여: " + balance.remain + "일");
  }
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sh = ss.getSheetByName(REQUEST_SHEET);
  const id = "LV-" + Utilities.formatDate(new Date(), TIMEZONE, "yyyyMMddHHmmss") + "-" + Math.floor(Math.random()*1000);
  sh.appendRow([id,new Date(),store,name,phone,leaveType,startDate,endDate,days,reason,"대기","",""]);
  return { ok:true, message:"신청이 접수되었습니다.", id:id };
}

function getRequests_() {
  const sh = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(REQUEST_SHEET);
  const values = sh.getDataRange().getValues();
  if (values.length <= 1) return [];
  const h = values[0];
  return values.slice(1).reverse().map(rowToObject_(h));
}

function getMyRequests_(name, phone) {
  const p = normalizePhone_(phone);
  return getRequests_().filter(r => String(r["이름"]) === String(name) && normalizePhone_(r["연락처"]) === p);
}

function getBalance_(name, phone) {
  const emp = findEmployee_(name, phone);
  if (!emp) return { registered:false, base:0, used:0, pending:0, remain:0, hireDate:"", workYears:0, workMonths:0 };
  const base = calculateAnnualLeave_(emp["입사일"]);
  let used = 0, pending = 0;
  const p = normalizePhone_(phone);
  getRequests_().forEach(r => {
    if (String(r["이름"]) === String(name) && normalizePhone_(r["연락처"]) === p) {
      if (r["상태"] === "승인") used += Number(r["사용일수"] || 0);
      if (r["상태"] === "대기") pending += Number(r["사용일수"] || 0);
    }
  });
  const svc = getServicePeriod_(emp["입사일"]);
  return { registered:true, base:base, used:used, pending:pending, remain:base-used, hireDate:formatDateOnly_(emp["입사일"]), workYears:svc.years, workMonths:svc.months };
}

function findEmployee_(name, phone) {
  const sh = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(EMPLOYEE_SHEET);
  const rows = sh.getDataRange().getValues();
  if (rows.length <= 1) return null;
  const h = rows[0], p = normalizePhone_(phone);
  for (let i=1;i<rows.length;i++) {
    const o = rowToObject_(h)(rows[i]);
    if (String(o["이름"]).trim() === String(name).trim() && normalizePhone_(o["연락처"]) === p) return o;
  }
  return null;
}

function updateStatus_(id, status, memo) {
  const sh = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(REQUEST_SHEET);
  const rows = sh.getDataRange().getValues();
  for (let i=1;i<rows.length;i++) {
    if (String(rows[i][0]) === String(id)) {
      sh.getRange(i+1,11).setValue(status);
      sh.getRange(i+1,12).setValue(memo || "");
      sh.getRange(i+1,13).setValue(new Date());
      return { ok:true, message:status + " 처리되었습니다." };
    }
  }
  throw new Error("해당 신청건을 찾을 수 없습니다.");
}

function calculateAnnualLeave_(hireDateValue) {
  const hire = parseDate_(hireDateValue), today = new Date();
  if (!hire || hire > today) return 0;
  const svc = getServicePeriod_(hire);
  if (svc.years < 1) return Math.min(Math.max(svc.totalMonths, 0), 11);
  return Math.min(15 + Math.floor((svc.years - 1) / 2), 25);
}

function getServicePeriod_(hireDateValue) {
  const hire = parseDate_(hireDateValue), today = new Date();
  let years = today.getFullYear() - hire.getFullYear();
  let months = today.getMonth() - hire.getMonth();
  if (today.getDate() < hire.getDate()) months--;
  if (months < 0) { years--; months += 12; }
  return { years:Math.max(0,years), months:Math.max(0,months), totalMonths:Math.max(0, years*12 + months) };
}

function parseDate_(v) {
  if (v instanceof Date) return v;
  if (!v) return null;
  const p = String(v).trim().split("-");
  if (p.length === 3) return new Date(Number(p[0]), Number(p[1])-1, Number(p[2]));
  const d = new Date(v);
  return isNaN(d.getTime()) ? null : d;
}
function formatDateOnly_(v) { const d = parseDate_(v); return d ? Utilities.formatDate(d, TIMEZONE, "yyyy-MM-dd") : ""; }
function normalizePhone_(p) { return String(p || "").replace(/[^0-9]/g, ""); }
function rowToObject_(h) {
  return function(row) {
    const o = {};
    h.forEach((key,i) => {
      let v = row[i];
      if (v instanceof Date) {
        if (String(key).indexOf("일") >= 0 && !["신청일시","처리일시","등록일시"].includes(String(key))) v = Utilities.formatDate(v, TIMEZONE, "yyyy-MM-dd");
        else v = Utilities.formatDate(v, TIMEZONE, "yyyy-MM-dd HH:mm");
      }
      o[key] = v;
    });
    return o;
  };
}
